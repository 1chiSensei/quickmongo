QuickMongo

Documentation

QuickMongo

Example

const { Database } = require("quickmongo");
const db = new Database("mongodb://localhost/quickmongo");

db.on("ready", () => {
    Console.log("Database connected!");
});

db.set("foo", "bar");

db.get("foo").then(console.log);
